      vi.advanceTimersByTime(1000);

    act(() => {
      vi.advanceTimersByTime(2000);
    });

    expect(result.current.secondsLeft).toBe(0);
    expect(result.current.isRunning).toBe(false);

    const { result } = renderHook(() =>
      useCountdown({
        initialSeconds: 2,
        autoStart: true,
        onFinish,
      }),
    );

    act(() => {
      vi.advanceTimersByTime(2000);
    });

    expect(result.current.secondsLeft).toBe(0);
    expect(onFinish).toHaveBeenCalledTimes(1);

    // 追加で進めても onFinish が増えないこと
    act(() => {
      vi.advanceTimersByTime(3000);
    });
    expect(onFinish).toHaveBeenCalledTimes(1);
  });

  it('reset で秒数と状態がリセットされる', () => {
    const { result } = renderHook(() =>
      useCountdown({
        initialSeconds: 5,
        autoStart: true,
      }),
    );

    act(() => {
      vi.advanceTimersByTime(2000);
    });
    expect(result.current.secondsLeft).toBe(3);

    act(() => {
      result.current.reset(10);
    });

    expect(result.current.secondsLeft).toBe(10);
    expect(result.current.isRunning).toBe(false);

    act(() => {
      result.current.start();
      vi.advanceTimersByTime(1000);
    });

    expect(result.current.secondsLeft).toBe(9);

## 1️⃣ {{sessionsRepo.ts}}（前提のおさらい）

こんな感じの構成になっている想定です：

```// src/lib/db/sessionsRepo.ts
import { getDB } from './openDB';
import type { SessionRecord } from '@/types/session';

const SESSION_STORE = 'sessions';

function generateId() {
  if (typeof crypto !== 'undefined' && 'randomUUID' in crypto) {
    return crypto.randomUUID();
  }
  return `session-${Date.now()}-${Math.random().toString(16).slice(2)}`;
}

export async function createSession(themeIds: string[]): Promise<SessionRecord> {
  const db = await getDB();
  const now = new Date().toISOString();

  const session: SessionRecord = {
    id: generateId(),
    themeIds,
    startedAt: now,
    endedAt: null,
    memoCount: 0,
  };

  const tx = db.transaction(SESSION_STORE, 'readwrite');
  await tx.store.add(session);
  await tx.done;

  return session;
}

export async function completeSession(
  sessionId: string,
  memoCount: number,
): Promise<void> {
  const db = await getDB();
  const tx = db.transaction(SESSION_STORE, 'readwrite');
  const store = tx.store;

  const existing = (await store.get(sessionId)) as SessionRecord | undefined;
  if (!existing) {
    console.warn('completeSession: session not found', sessionId);
    await tx.done;
    return;
  }

  const updated: SessionRecord = {
    ...existing,
    memoCount,
    endedAt: new Date().toISOString(),
  };

  await store.put(updated);
  await tx.done;
}

export async function getAllSessions(): Promise<SessionRecord[]> {
  const db = await getDB();
  const tx = db.transaction(SESSION_STORE, 'readonly');
  const all = await tx.store.getAll();
  await tx.done;
  return (all ?? []) as SessionRecord[];
}

export async function getSessionById(
  sessionId: string,
): Promise<SessionRecord | undefined> {
  const db = await getDB();
  const tx = db.transaction(SESSION_STORE, 'readonly');
  const record = (await tx.store.get(sessionId)) as SessionRecord | undefined;
  await tx.done;
  return record;
}
```

（だいたいこのイメージであれば OK です。フィールド名が多少違っていてもテスト側を合わせれば動きます）

// sessionsRepo から見ると './openDB' をインポートしているので、
// テストファイルからは '../openDB' をモックする

    __reset();

  it('createSession で新しいセッションを作成し、getAllSessions で取得できる', async () => {

    expect(stored.themeIds).toEqual(themeIds);
  });

  it('getSessionById で特定のセッションを取得できる', async () => {

## 前提：memosRepo のざっくり形

こんな感じの実装を想定しています（多少違っていてもテスト側を合わせればOKです）。

```// src/lib/db/memosRepo.ts
import { MemoRecord } from '@/types/memo';
import { getDB } from './openDB';

const MEMO_STORE = 'memos';

function generateId() {
  if (typeof crypto !== 'undefined' && 'randomUUID' in crypto) {
    return crypto.randomUUID();
  }
  return `memo-${Date.now()}-${Math.random().toString(16).slice(2)}`;
}

/**
 * メモを保存（同じ id があれば上書き）
 */
export async function saveMemo(
  memo: Omit<MemoRecord, 'id' | 'createdAt' | 'updatedAt'> & {
    id?: string;
  },
): Promise<MemoRecord> {
  const db = await getDB();
  const now = new Date().toISOString();

  const record: MemoRecord = {
    id: memo.id ?? generateId(),
    createdAt: now,
    updatedAt: now,
    ...memo,
  };

  const tx = db.transaction(MEMO_STORE, 'readwrite');
  await tx.store.put(record);
  await tx.done;

  return record;
}

/**
 * 指定セッションに紐づくメモをすべて取得する
 * - order 昇順で返す
 */
export async function getMemosBySessionId(
  sessionId: string,
): Promise<MemoRecord[]> {
  const db = await getDB();
  const tx = db.transaction(MEMO_STORE, 'readonly');
  const store = tx.store;

  const all = (await store.getAll()) as MemoRecord[];
  await tx.done;

  const filtered = (all ?? []).filter(memo => memo.sessionId === sessionId);
  filtered.sort((a, b) => a.order - b.order);

  return filtered;
}
```

    const s1Memos = await getMemosBySessionId('s1');
    const s2Memos = await getMemosBySessionId('s2');
    const s3Memos = await getMemosBySessionId('s3'); // 存在しない

    expect(s1Memos).toHaveLength(2);
    expect(s2Memos).toHaveLength(1);
    expect(s3Memos).toHaveLength(0);

    const s1Orders = s1Memos.map(m => m.order);
    // order 昇順で並んでいること（1,2）
    expect(s1Orders).toEqual([1, 2]);
    expect(s1Memos[0].textContent).toBe('s1-1');
    expect(s1Memos[1].textContent).toBe('s1-2');
  });

  it('メモがひとつも無いセッションIDの場合は空配列を返す', async () => {
    const memos = await getMemosBySessionId('no-memo-session');
    expect(Array.isArray(memos)).toBe(true);
    expect(memos).toHaveLength(0);
  });
});
