## 📄 {{src/app/session/complete/page.tsx}}

```'use client';

import React, { useEffect, useState } from 'react';
import { useSearchParams } from 'next/navigation';

import { Button } from '@/components/ui/Button';
import { getSessionById } from '@/lib/db/sessionsRepo';
import type { SessionRecord } from '@/types/session';

type LoadStage = 'idle' | 'loading' | 'loaded' | 'error';

export default function SessionCompletePage() {
  const searchParams = useSearchParams();
  const sessionId = searchParams.get('sessionId');

  const [stage, setStage] = useState<LoadStage>('idle');
  const [session, setSession] = useState<SessionRecord | null>(null);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const load = async () => {
      // sessionId が無い場合はエラー扱いにする
      if (!sessionId) {
        setError('セッションIDが指定されていません。');
        setStage('error');
        return;
      }

      try {
        setStage('loading');
        setError(null);

        const result = await getSessionById(sessionId);

        if (!result) {
          setError('対象のセッションが見つかりませんでした。');
          setStage('error');
          return;
        }

        setSession(result);
        setStage('loaded');
      } catch (e) {
        console.error('Failed to load session', e);
        setError('セッション情報の読み込み中にエラーが発生しました。');
        setStage('error');
      }
    };

    void load();
  }, [sessionId]);

  // ローディング中
  if (stage === 'loading' || stage === 'idle') {
    return (
      <main className="mx-auto max-w-3xl px-4 py-8">
        <h1 className="text-xl font-semibold text-slate-900">
          セッション結果を読み込んでいます…
        </h1>
        <p className="mt-2 text-sm text-slate-600">
          少しお待ちください。
        </p>
      </main>
    );
  }

  // エラー時（sessionId 不正・存在しない 等）
  if (stage === 'error' || !session) {
    return (
      <main className="mx-auto max-w-3xl px-4 py-8">
        <h1 className="text-xl font-semibold text-slate-900">
          セッションの結果を表示できません
        </h1>
        {error && (
          <p className="mt-2 text-sm text-slate-600">
            {error}
          </p>
        )}
        <div className="mt-6 flex flex-wrap gap-3">
          <Button href="/session" variant="primary">
            新しくセッションを始める
          </Button>
          <Button href="/history" variant="secondary">
            履歴一覧を見る
          </Button>
          <Button href="/" variant="ghost">
            トップへ戻る
          </Button>
        </div>
      </main>
    );
  }

  // 正常ケース：セッションが取得できた
  const memoCount = session.memoCount;
  const themeCount = session.themeIds.length;
  const started = session.startedAt
    ? new Date(session.startedAt)
    : null;
  const ended = session.endedAt ? new Date(session.endedAt) : null;

  const startedLabel = started
    ? started.toLocaleString()
    : '不明';
  const endedLabel = ended ? ended.toLocaleString() : '進行中';

  // 所要時間（分）のざっくり表示（endedAt がある場合のみ）
  let durationMinutes: number | null = null;
  if (started && ended) {
    const diffMs = ended.getTime() - started.getTime();
    if (diffMs > 0) {
      durationMinutes = Math.round(diffMs / 1000 / 60);
    }
  }

  return (
    <main className="mx-auto max-w-3xl px-4 py-8">
      {/* ヘッダー */}
      <header className="border-b border-slate-200 pb-4">
        <h1 className="text-2xl font-semibold text-slate-900">
          セッションが完了しました
        </h1>
        <p className="mt-2 text-sm text-slate-600">
          おつかれさまでした。今回のセッションで{' '}
          <span className="font-semibold text-slate-900">
            {memoCount}
          </span>
          件のメモを書き出しました。
        </p>
      </header>

      {/* セッション概要 */}
      <section className="mt-6 rounded-lg border border-slate-200 bg-white p-4 shadow-sm">
        <h2 className="text-sm font-medium text-slate-800">
          セッション概要
        </h2>
        <dl className="mt-3 grid grid-cols-1 gap-2 text-sm text-slate-700 sm:grid-cols-2">
          <div className="flex flex-col">
            <dt className="text-xs text-slate-500">開始時刻</dt>
            <dd>{startedLabel}</dd>
          </div>
          <div className="flex flex-col">
            <dt className="text-xs text-slate-500">終了時刻</dt>
            <dd>{endedLabel}</dd>
          </div>
          <div className="flex flex-col">
            <dt className="text-xs text-slate-500">テーマ数</dt>
            <dd>{themeCount} 件</dd>
          </div>
          <div className="flex flex-col">
            <dt className="text-xs text-slate-500">メモ数</dt>
            <dd>{memoCount} 件</dd>
          </div>
          {durationMinutes !== null && (
            <div className="flex flex-col">
              <dt className="text-xs text-slate-500">所要時間（概算）</dt>
              <dd>{durationMinutes} 分</dd>
            </div>
          )}
        </dl>
      </section>

      {/* アクションボタン */}
      <section className="mt-8 flex flex-wrap gap-3">
        <Button href={`/history/${session.id}`} variant="primary">
          このセッションの詳細を見る
        </Button>
        <Button href="/history" variant="secondary">
          履歴一覧を見る
        </Button>
        <Button href="/session" variant="ghost">
          もう一度セッションを始める
        </Button>
      </section>
    </main>
  );
}
```

## 🔍 使い方の前提メモ

- セッション画面からは、セッション完了時に：

```router.push(`/session/complete?sessionId=${sessionId}`);

```

のように遷移する想定です。

- もし「クエリを使わず、*最新の完了セッションを出したい*」という方向に変えるなら、
** {{getLatestFinishedSession()}} みたいな Repo を作って、
** {{sessionId}} なしのときはそれを読むように変更すればOKです。
