いいですね、これで一通り「やったセッションがちゃんと溜まっていく感じ」が出せます。
{{/history/page.tsx}} *の骨組みコード*を用意します。

- 前提
** {{getAllSessions}}（{{src/lib/db/sessionsRepo.ts}}）が実装済み
** {{Button}} コンポーネントが {{@/components/ui/Button}} にある
** {{SessionCard}} を使うか悩むところですが、骨組みとしてはまず*ページ内で簡単なカード表示*を直接書いておきます（あとで分離しやすい構造にしておきます）

----

## 📄 {{src/app/history/page.tsx}}（履歴一覧の骨組み）

```'use client';

import React, { useEffect, useMemo, useState } from 'react';

import { Button } from '@/components/ui/Button';
import { getAllSessions } from '@/lib/db/sessionsRepo';
import type { SessionRecord } from '@/types/session';

type LoadStage = 'idle' | 'loading' | 'loaded' | 'error';

export default function HistoryPage() {
  const [stage, setStage] = useState<LoadStage>('idle');
  const [sessions, setSessions] = useState<SessionRecord[]>([]);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    const load = async () => {
      try {
        setStage('loading');
        setError(null);

        const all = await getAllSessions();
        setSessions(all);
        setStage('loaded');
      } catch (e) {
        console.error('Failed to load sessions', e);
        setError('履歴の読み込み中にエラーが発生しました。');
        setStage('error');
      }
    };

    void load();
  }, []);

  // 新しい順にソート
  const sortedSessions = useMemo(() => {
    const copy = [...sessions];
    copy.sort((a, b) => {
      const aKey = a.endedAt ?? a.startedAt;
      const bKey = b.endedAt ?? b.startedAt;
      const aTime = aKey ? new Date(aKey).getTime() : 0;
      const bTime = bKey ? new Date(bKey).getTime() : 0;
      return bTime - aTime; // 降順
    });
    return copy;
  }, [sessions]);

  // ローディング中
  if (stage === 'idle' || stage === 'loading') {
    return (
      <main className="mx-auto max-w-3xl px-4 py-8">
        <h1 className="text-xl font-semibold text-slate-900">
          履歴を読み込んでいます…
        </h1>
        <p className="mt-2 text-sm text-slate-600">
          これまでのセッションの記録を取得しています。
        </p>
      </main>
    );
  }

  // エラー時
  if (stage === 'error') {
    return (
      <main className="mx-auto max-w-3xl px-4 py-8">
        <h1 className="text-xl font-semibold text-slate-900">
          履歴を表示できません
        </h1>
        {error && (
          <p className="mt-2 text-sm text-slate-600">
            {error}
          </p>
        )}
        <div className="mt-6 flex flex-wrap gap-3">
          <Button href="/session" variant="primary">
            新しくセッションを始める
          </Button>
          <Button href="/" variant="secondary">
            トップへ戻る
          </Button>
        </div>
      </main>
    );
  }

  const hasSessions = sortedSessions.length > 0;

  return (
    <main className="mx-auto max-w-3xl px-4 py-8">
      {/* ヘッダー */}
      <header className="border-b border-slate-200 pb-4">
        <h1 className="text-2xl font-semibold text-slate-900">
          履歴一覧
        </h1>
        <p className="mt-2 text-sm text-slate-600">
          これまでに行ったセッションの記録です。
        </p>
      </header>

      {/* 空状態 */}
      {!hasSessions && (
        <section className="mt-8 rounded-lg border border-dashed border-slate-300 bg-slate-50 p-6 text-center">
          <p className="text-sm text-slate-600">
            まだセッション履歴がありません。
          </p>
          <p className="mt-2 text-xs text-slate-500">
            セッションを開始すると、ここに記録が表示されます。
          </p>
          <div className="mt-4 flex justify-center">
            <Button href="/session" variant="primary">
              セッションを始める
            </Button>
          </div>
        </section>
      )}

      {/* 一覧表示 */}
      {hasSessions && (
        <section className="mt-6">
          <ul className="flex flex-col gap-3">
            {sortedSessions.map(session => {
              const started = session.startedAt
                ? new Date(session.startedAt)
                : null;
              const ended = session.endedAt
                ? new Date(session.endedAt)
                : null;

              const startedLabel = started
                ? started.toLocaleString()
                : '不明';
              const endedLabel = ended
                ? ended.toLocaleString()
                : '未完了';

              return (
                <li
                  key={session.id}
                  className="rounded-lg border border-slate-200 bg-white p-4 shadow-sm"
                >
                  <div className="flex flex-col gap-2 sm:flex-row sm:items-center sm:justify-between">
                    <div>
                      <p className="text-xs font-medium text-slate-500">
                        {startedLabel} 開始
                        {session.endedAt && ` / ${endedLabel} 終了`}
                      </p>
                      <p className="mt-1 text-sm font-semibold text-slate-900">
                        テーマ {session.themeIds.length} 件・メモ{' '}
                        {session.memoCount} 件
                      </p>
                    </div>
                    <div className="mt-2 flex gap-2 sm:mt-0">
                      <Button
                        href={`/history/${session.id}`}
                        variant="secondary"
                        size="sm"
                      >
                        詳細を見る
                      </Button>
                    </div>
                  </div>
                </li>
              );
            })}
          </ul>
        </section>
      )}

      {/* フッター導線 */}
      <section className="mt-8 flex flex-wrap gap-3 border-t border-slate-200 pt-4">
        <Button href="/session" variant="primary">
          新しくセッションを始める
        </Button>
        <Button href="/" variant="secondary">
          トップへ戻る
        </Button>
      </section>
    </main>
  );
}
```

- まずは*ページ内でカード表示*まで完結させておき、
UI を共通化したくなったら後から {{SessionCard}} コンポーネントに切り出せばOKです。
- {{variant}} / {{size}} は、あなたが定義済みの {{Button}} props に合わせて調整してください（ここは仮）。
